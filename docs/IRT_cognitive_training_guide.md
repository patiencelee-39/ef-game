function fisherInformation(theta, item) {
  const prob = calculateProbability(theta, item);
  const {a, c} = item;
  const numerator = a * a * (prob - c) * (prob - c) * (1 - prob);
  const denominator = prob * (1 - c) * (1 - c);
  return numerator / denominator;
}
